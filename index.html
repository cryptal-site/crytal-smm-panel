    /* styles.css */ body { font-family: Arial, sans-serif; margin: 20px; } .container { max-width: 600px; margin: auto; background: #f4f4f4; padding: 20px; border-radius: 5px; } #taskList { list-style-type: none; padding: 0; } #taskList li { background: #fff; margin: 5px 0; padding: 10px; border: 1px solid #ddd; border-radius: 3px; } /* script.js */ const taskInput = document.getElementById('taskInput'); const addTaskBtn = document.getElementById('addTaskBtn'); const taskList = document.getElementById('taskList'); const tasks = JSON.parse(localStorage.getItem('tasks')) || []; function renderTasks() { taskList.innerHTML = ''; tasks.forEach((task, index) => { const li = document.createElement('li'); li.innerText = task; taskList.appendChild(li); li.addEventListener('click', () => { tasks.splice(index, 1); saveTasks(); }); }); } function saveTasks() { localStorage.setItem('tasks', JSON.stringify(tasks)); renderTasks(); } addTaskBtn.addEventListener('click', () => { const task = taskInput.value.trim(); if (task) { tasks.push(task); taskInput.value = ''; saveTasks(); } }); renderTasks();
